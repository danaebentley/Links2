#This file contains information on aliases that I find useful in regular use

#Shortprompt removes most of the dir tree from the prompt
#Remember that you can use pwd to see entire path to current folder!
alias shortprompt="PS1='\u:\W\$ '"

#kwweather shows a weather forecast at KW Airport
alias kwweather="curl wttr.in/ykf"

#lndnweather shows a weather forecast for London ON airport
alias lndnweather="curl wttr.in/yxu"
